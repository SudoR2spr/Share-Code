    <script>
        document.addEventListener('DOMContentLoaded', function() {
            const apiKey = 'YOUR_PIXABAY_API_KEY'; // Replace with your actual Pixabay API key
            const searchInput = document.getElementById('search-input');
            const searchBtn = document.getElementById('search-btn');
            const gallery = document.getElementById('gallery');
            
            // Function to fetch images from Pixabay API
            async function fetchImages(query) {
                gallery.innerHTML = '<div class="loading">Loading beautiful images for you... <span class="emoji">‚è≥</span></div>';
                
                try {
                    const response = await fetch(`https://pixabay.com/api/?key=${apiKey}&q=${encodeURIComponent(query)}&image_type=photo&pretty=true&per_page=20`);
                    const data = await response.json();
                    
                    if (data.hits.length === 0) {
                        gallery.innerHTML = '<div class="loading">No images found. Try a different search! <span class="emoji">üòï</span></div>';
                        return;
                    }
                    
                    gallery.innerHTML = '';
                    
                    data.hits.forEach(image => {
                        const imageCard = document.createElement('div');
                        imageCard.className = 'image-card';
                        
                        imageCard.innerHTML = `
                            <img src="${image.webformatURL}" alt="${image.tags}">
                            <div class="image-info">
                                <h3>${image.tags.split(',')[0]}</h3>
                                <p>By ${image.user}</p>
                                <button class="download-btn" onclick="window.open('${image.largeImageURL}', '_blank')">
                                    <i>‚¨áÔ∏è</i> Download HD
                                </button>
                            </div>
                        `;
                        
                        gallery.appendChild(imageCard);
                    });
                } catch (error) {
                    gallery.innerHTML = '<div class="loading">Oops! Something went wrong. Please try again later. <span class="emoji">üò¢</span></div>';
                    console.error('Error fetching images:', error);
                }
            }
            
            // Event listeners
            searchBtn.addEventListener('click', () => {
                const query = searchInput.value.trim();
                if (query) {
                    fetchImages(query);
                }
            });
            
            searchInput.addEventListener('keypress', (e) => {
                if (e.key === 'Enter') {
                    const query = searchInput.value.trim();
                    if (query) {
                        fetchImages(query);
                    }
                }
            });
            
            // Load some default images on first load
            fetchImages('nature');
        });
    </script>
